server {
        listen 443 ssl http2;

	root /home/www/exs.lv;
        index index.php;

        server_name exs.lv coding.lv www.exs.lv rp.exs.lv www.coding.lv rs.exs.lv runescape.exs.lv lol.exs.lv static.exs.lv secure.exs.lv;
        client_max_body_size 60M;

        #SSL
        ssl_certificate /etc/apache2/ssl_2015/ssl-bundle.crt;
        ssl_certificate_key /etc/apache2/ssl_2015/exs.key;
	ssl_protocols        SSLv3 TLSv1 TLSv1.1 TLSv1.2;	# default on newer versions
	ssl_ciphers          ECDHE-RSA-AES256-GCM-SHA384:ECDHE-RSA-AES128-GCM-SHA256:ECDHE-RSA-RC4-SHA:ECDHE-RSA-AES128-SHA:RC4-SHA:RC4-MD5:ECDHE-RSA-AES256-SHA:AES256-SHA:ECDHE-RSA-DES-CBC3-SHA:DES-CBC3-SHA:AES128-SHA;
	ssl_prefer_server_ciphers on;

        location / {

	        if (-f $request_filename) {
			break;
		}
        	if (!-f $request_filename) {
			rewrite ^/(.+)$ /index.php?fakeurl=$1 last;
			break;
        	}

        }

        location /api {
		return 301 https://exs.lv/api/;
        }

        location /api/ {
		rewrite ^/api/(.*)$ /modules/api/index.php?params=$1 last;
        }

        location ~ /dati/bildes/topic-av/(.*\.jpg) {
                expires 1d;
                try_files $uri /index.php?viewcat=page-avatars&img=$1;
        }

        location ~* \.(png|jpg|jpeg|gif|ico)$ {
                expires 1y;
        }

        location /js/ {
		expires 3h;
                rewrite ^/js/(.*\.js) /combine.php?type=javascript&files=$1;
        }

        location /css/ {
		expires 3h;
                rewrite ^/css/(.*\.css) /combine.php?type=css&files=$1;
        }

        location ~ \.php$ {
               include snippets/fastcgi-php.conf;
               fastcgi_pass unix:/var/run/php5-fpm.sock;
        }

        location ~ /\.ht {
                deny all;
        }

        location ~ /\.tpl {
                deny all;
        }

}
