backend default {
    .host = "127.0.0.1";
    .port = "8080";
}

sub vcl_recv {
  if ((req.request == "GET" && req.url ~ "\.(png|jpg|gif|ico|css|js)$")) {
       remove req.http.Cookie;
   }
}
sub vcl_deliver {
        if (obj.hits > 0) {
                set resp.http.X-Cache = "HIT";
        } else {
                set resp.http.X-Cache = "MISS";
        }
}
