<?php
set_time_limit(600);

$maps = $db->get_results("SELECT * FROM lostmaps ORDER BY rand() LIMIT 100");


/* load libraries */
require_once('includes_3.0/class.upload.php');

foreach($maps as $map) {
  $map->title = strtolower($map->title);
	$link = "http://images.google.com/images?hl=en&q=".$map->title."+map+filetype:jpg&btnG=Bilder-Suche&gbv=1";
	$code = file_get_contents($link,'r');

	ereg ("imgurl=http://www.[A-Za-z0-9-]*.[A-Za-z]*[^.]*.[A-Za-z]*", $code, $img);

	ereg ("http://(.*)", $img[0], $img_pic);

	if($img = file_get_contents($img_pic[0])) {
	  $file = 'tmpimg/large/'.$map->title;
		file_put_contents($file,$img);
		echo 'processing '.$map->title.'<br />';



		$foo = new Upload($file);
		$foo->file_new_name_body = $map->title;
	  $foo->image_resize = true;
	  $foo->image_convert = 'jpg';
	  $foo->image_x = 128;
	  $foo->image_y =	79;
	  $foo->allowed = array('image/*');
		$foo->image_ratio_crop = true;
		$foo->jpeg_quality = 96;
		$foo->file_overwrite = true;
	  $foo->process('tmpimg/ready/');

	  if ($foo->processed) {
		  $db->query("DELETE FROM lostmaps WHERE title = '".sanitize($map->title)."'");
		}


	}

}


?>

<meta HTTP-EQUIV="REFRESH" content="2; url=/dev/" />

<?php

exit;
